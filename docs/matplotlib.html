5.1 line plot

import matplotlib.pyplot as plt
year = [1950, 1970, 1990, 2010]
pop = [2.519, 3.692, 5.263, 6.972] #pop = population, 單位: billion
plt.plot(year, pop) #只寫到這邊，不會畫圖，因為他還不知道你要不要加title那些
plt.show() #寫show的話，他才知道你指令都輸入完了，才會畫圖
